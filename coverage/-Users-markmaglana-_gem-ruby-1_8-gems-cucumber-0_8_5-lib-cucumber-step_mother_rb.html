291     def best_matches(step_name, step_matches) #:nodoc:
 
292       no_groups      = step_matches.select {|step_match| step_match.args.length == 0}
 
293       max_arg_length = step_matches.map {|step_match| step_match.args.length }.max
 
294       top_groups     = step_matches.select {|step_match| step_match.args.length == max_arg_length }
 
295 
296       if no_groups.any?
 
297         longest_regexp_length = no_groups.map {|step_match| step_match.text_length }.max
 
298         no_groups.select {|step_match| step_match.text_length == longest_regexp_length }
 
299       elsif top_groups.any?
 
300         shortest_capture_length = top_groups.map {|step_match| step_match.args.inject(0) {|sum, c| sum + c.to_s.length } }.min
 
301         top_groups.select {|step_match| step_match.args.inject(0) {|sum, c| sum + c.to_s.length } == shortest_capture_length }
